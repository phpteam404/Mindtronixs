LEFT JOIN `user_role` `ur` ON `u`.`user_role_id`=`ur`.`id`

AND `u`.`franchise_id` = '1'
AND `u`.`user_status` IN(0, 1)
ORDER BY `u`.`id` DESC
 LIMIT 10
